## 一、意图识别：三路并行 + 加权投票

**文件**：`core/intent_recognizer.py` 第 125-162 行

```python
async def recognize(self, message: str, history=None) -> IntentResult:
    key = self._cache_key(message)
    if key in self._cache:
        self.cache_hits += 1
        return self._cache[key]
    self.cache_misses += 1

    t0 = time.monotonic()

    # ① LLM 和 Embedding 并行（Embedding 不可用时跳过）
    llm_task = asyncio.create_task(self._llm_recognize(message, history))
    emb_task = asyncio.create_task(self._embedding_recognize(message)) if self._embedding_enabled else None
    pat      = self._pattern_recognize(message)  # ② 同步执行，零延迟

    if emb_task:
        llm, emb = await asyncio.gather(llm_task, emb_task)  # ③ 并行等待
    else:
        llm = await llm_task
        emb = {"intent": IntentCategory.OTHER, "confidence": 0.0}

    intent = self._vote(llm, emb, pat)  # ④ 加权投票合并
    entities = await self._extract_entities(message)
    urgency  = self._urgency(message, intent)

    result = IntentResult(...)

    # ⑤ LRU 缓存：超 1000 条时淘汰最旧一半
    if len(self._cache) >= 1000:
        for k in list(self._cache)[:500]:
            del self._cache[k]
    self._cache[key] = result
    return result
```

### 逐行解析

| 标记 | 设计意图 |
|------|----------|
| ① | `asyncio.create_task` 立即启动协程但不等待，实现真正的并行 |
| ② | Pattern 匹配是纯 CPU 操作（字符串匹配），不需要异步，直接同步执行 |
| ③ | `asyncio.gather` 同时等待 LLM 和 Embedding，总延迟 = max(两者延迟)，而非相加 |
| ④ | 三路结果各自带置信度，乘以权重后累加，取最高分的意图 |
| ⑤ | 简单 LRU：dict 在 Python 3.7+ 保持插入顺序，删前一半即淘汰最旧的 |

### 加权投票实现

```python
def _vote(self, llm, emb, pat) -> IntentCategory:
    if self._embedding_enabled:
        weights = [(llm, 0.7), (emb, 0.2), (pat, 0.1)]
    else:
        weights = [(llm, 0.85), (pat, 0.15)]  # 动态调整权重

    scores = {}
    for result, w in weights:
        cat  = result.get("intent", IntentCategory.OTHER)
        conf = result.get("confidence", 0.0)
        scores[cat] = scores.get(cat, 0.0) + w * conf

    best = max(scores, key=scores.get)
    return best if scores[best] >= self.threshold else IntentCategory.OTHER
```

**关键点**：当 Embedding 不可用（第三方 API）时，权重自动重分配为 LLM 85% + Pattern 15%，不需要改调用逻辑。

## 二、MCP 工具调用：查询改写 + 重排

**文件**：`mcp/tool_manager.py` 第 252-290 行

```python
async def search_with_rewrite(self, tool_name, query, top_k=5, context=None) -> ToolResult:
    """完整的检索优化链路"""

    # ① 查询改写：LLM 生成多角度子查询
    sub_queries = await self.rewrite_query(query, n=3)
    # 例如: "退款流程" → ["如何申请退款", "退款需要多少天", "退款政策是什么"]

    # ② 并行召回：所有子查询同时检索
    tasks = [
        self.call(tool_name, {"query": q}, context, use_cache=True)
        for q in sub_queries
    ]
    results = await asyncio.gather(*tasks, return_exceptions=True)

    # ③ 合并去重（按内容哈希去重）
    seen, merged = set(), []
    for r in results:
        if isinstance(r, ToolResult) and r.success and isinstance(r.data, list):
            for item in r.data:
                key = hashlib.md5(str(item).encode()).hexdigest()
                if key not in seen:
                    seen.add(key)
                    merged.append(item)

    # ④ LLM 重排：按语义相关性重新排序，取 Top-K
    reranked = await self._rerank(query, merged, top_k)
    return ToolResult(success=True, data=reranked, tool_name=tool_name, reranked=True)
```

### 逐步解析

| 步骤 | 解决的问题 | 原理 |
|------|-----------|------|
| ① 查询改写 | 召回不全 | 单一查询只覆盖一个角度，改写为 3 个角度后覆盖面 ×3 |
| ② 并行召回 | 延迟控制 | 3 个子查询同时发出，总延迟 ≈ 单次延迟 |
| ③ 合并去重 | 结果重复 | 不同子查询可能召回相同文档，用 MD5 哈希去重 |
| ④ LLM 重排 | 排序不好 | 向量相似度 ≠ 对用户有用，LLM 能理解真正的语义相关性 |

### 熔断器实现

```python
class CircuitBreaker:
    """三态熔断器：CLOSED → OPEN → HALF_OPEN → CLOSED"""

    def allow(self) -> bool:
        if self.state == CircuitState.CLOSED:
            return True  # 正常放行
        if self.state == CircuitState.OPEN:
            # 超过恢复时间后进入探测状态
            if time.monotonic() - self.opened_at >= self.recovery_s:
                self.state = CircuitState.HALF_OPEN
                return True  # 放行一次探测
            return False  # 拒绝请求
        return True  # HALF_OPEN：放行探测

    def record_failure(self) -> None:
        self.fail_count += 1
        if self.fail_count >= self.threshold:
            self.state = CircuitState.OPEN
            self.opened_at = time.monotonic()
            # 连续失败 5 次 → 自动断开 60s
```

**关键点**：熔断器是工具级别的，每个工具独立一个。工具 A 熔断不影响工具 B。

## 三、记忆管理：自动压缩机制

**文件**：`memory/conversation_memory.py` 第 113-136 行

```python
async def add_message(self, user_id, conv_id, role, content, metadata=None):
    """写入工作记忆，超阈值时自动压缩。"""
    msg = Message(role=role, content=content, metadata=metadata or {})
    key = self._wm_key(user_id, conv_id)

    # ① 追加到 Redis 列表（左推，最新在前）
    self._redis.lpush(key, json.dumps({
        "role": msg.role.value,
        "content": msg.content,
        "ts": msg.timestamp.isoformat(),
        "metadata": msg.metadata,
    }))
    self._redis.expire(key, 86400)  # ② 24h TTL，过期自动清理

    # ③ 超过压缩阈值时触发压缩
    if self._redis.llen(key) >= self.COMPRESS_AT:
        await self._compress(user_id, conv_id)
```

### 压缩流程

```python
async def _compress(self, user_id, conv_id):
    messages = await self._get_working_memory(user_id, conv_id)
    to_compress = messages[:-5]   # 保留最近 5 条
    keep        = messages[-5:]

    # ① LLM 生成摘要
    text = "\n".join(f"{m.role.value}: {m.content}" for m in to_compress)
    prompt = f"用 2-3 句话总结以下对话的关键信息：\n{text}"
    resp = await self._client.messages.create(model=self._model, ...)
    summary = resp.content[0].text.strip()

    # ② 摘要存 Redis（累加到旧摘要后面）
    old_summary = self._redis.get(summary_key) or ""
    self._redis.setex(summary_key, 86400, f"{old_summary}\n{summary}".strip())

    # ③ 旧消息存入情景记忆（ChromaDB），供跨会话检索
    await self._store_episodic(user_id, conv_id, text, summary)

    # ④ 重置工作记忆为最近 5 条
    self._redis.delete(key)
    for m in reversed(keep):
        self._redis.lpush(key, ...)
```

### 设计要点

| 设计 | 原因 |
|------|------|
| 保留最近 5 条 | 保证当前对话的连贯性，Agent 能看到最近上下文 |
| 摘要累加 | 多次压缩后摘要叠加，形成完整的会话历史线索 |
| 存入情景记忆 | 压缩不等于丢弃，历史信息通过语义检索仍可找回 |
| 24h TTL | 工作记忆是临时的，超过 24h 未活跃的会话自动清理 |

## 四、Agent 路由：三层决策 + 性能感知

**文件**：`agents/agent_orchestrator.py` 第 225-258 行

```python
async def run(self, req: Request) -> OrchestratorResult:
    t0 = time.monotonic()

    # ① 意图识别
    if req.intent is None:
        intent_result = await self._intent_recognizer.recognize(req.message)
        req.intent  = intent_result.intent
        req.urgency = intent_result.urgency

    # ② 路由：三层决策
    agent_type = self._route(req.intent, req.urgency)

    # ③ 执行（含降级）
    response = await self._execute(req, agent_type)

    # ④ 升级检查
    escalated = False
    if response.escalate or req.urgency == UrgencyLevel.CRITICAL:
        escalated = True
```

### 路由决策实现

```python
def _route(self, intent, urgency) -> AgentType:
    # 第一层：紧急度覆盖
    if urgency == UrgencyLevel.CRITICAL:
        return AgentType.ESCALATION

    # 第二层：意图映射
    if intent and intent in self._INTENT_ROUTING:
        target = self._INTENT_ROUTING[intent]
        if target in self._pool and self._pool[target]:
            return target

    # 第三层：默认降级
    return AgentType.GENERAL
```

### 性能路由（同类多实例时）

```python
def _best_agent(self, agent_type) -> Optional[BaseAgent]:
    """从同类 Agent 中选 routing_score 最高的。"""
    agents = self._pool.get(agent_type, [])
    if not agents:
        return None
    return max(agents, key=lambda a: a.stats.routing_score())

# routing_score 的计算
def routing_score(self) -> float:
    latency_score = 1.0 / (1.0 + self.avg_ms / 1000)
    return self.success_rate * 0.7 + latency_score * 0.3
```

**关键点**：`routing_score` 是动态的——每次请求后 `AgentStats` 实时更新，下次路由时自动选择表现更好的实例。这就是 Monitor 闭环的核心。

### 降级执行

```python
async def _execute(self, req, agent_type) -> AgentResponse:
    agent = self._best_agent(agent_type)
    if agent is None:
        agent = self._best_agent(AgentType.GENERAL)  # 降级

    response = await agent.handle(req)

    # 专属 Agent 失败时降级到 GeneralAgent
    if not response.success and agent_type != AgentType.GENERAL:
        logger.warning(f"{agent_type.value} 失败，降级到 GeneralAgent")
        fallback = self._best_agent(AgentType.GENERAL)
        if fallback:
            response = await fallback.handle(req)

    return response
```

**设计哲学**：宁可给用户一个通用但正确的回答，也不要因为专属 Agent 故障而返回错误。

## 五、并行协作：多 Agent 同时处理

**文件**：`agents/agent_orchestrator.py` 第 260-275 行

```python
async def run_parallel(self, req: Request, agent_types: List[AgentType]) -> OrchestratorResult:
    """并行派发给多个 Agent，合并结果。"""
    t0 = time.monotonic()

    # ① 所有 Agent 同时启动
    tasks = [self._execute(req, at) for at in agent_types]
    responses = await asyncio.gather(*tasks, return_exceptions=True)

    # ② 合并所有成功响应
    parts = []
    for r in responses:
        if isinstance(r, AgentResponse) and r.success:
            parts.append(f"[{r.agent_type.value}]\n{r.content}")

    combined = "\n\n".join(parts) if parts else "抱歉，所有 Agent 均处理失败。"
```

**使用场景**：用户说"我的订单出错了，而且被多扣了钱"——同时涉及技术问题和账单问题，可以并行派发给 TechnicalAgent 和 BillingAgent，各自处理后合并返回。

## 六、在线学习：从反馈中持续改进

**文件**：`core/intent_recognizer.py` 第 164-170 行

```python
def learn(self, message: str, correct: IntentCategory) -> None:
    """在线学习：将纠正样本加入模板，清除对应 Embedding 缓存。"""
    tpls = _TEMPLATES.setdefault(correct, [])
    if message not in tpls:
        tpls.append(message)
        self._tpl_embeddings.pop(correct, None)  # 下次重新计算
        logger.info(f"学习新样本 → {correct.value}: {message[:40]}")
```

### 工作原理

```
用户消息: "网页打不开了"
系统预测: QUERY（错误）
人工纠正: TECHNICAL（正确）
    ↓
learn("网页打不开了", IntentCategory.TECHNICAL)
    ↓
1. "网页打不开了" 加入 TECHNICAL 的模板列表
2. 清除 TECHNICAL 的 Embedding 缓存
3. 下次有类似消息时：
   - LLM 的 Few-shot 示例里多了这条
   - Embedding 匹配时会重新计算（包含新样本）
   - Pattern 匹配不受影响（需要手动加关键词）
```

**设计优势**：不需要重新训练模型，不需要重启服务，实时生效。

## 七、Unicode 安全：防止编码崩溃

**文件**：`agents/agent_orchestrator.py` `_call_llm` 方法

```python
async def _call_llm(self, req: Request) -> str:
    def _clean(s: str) -> str:
        return s.encode("utf-8", errors="ignore").decode("utf-8")

    messages = []
    if req.context:
        messages.append({"role": "user", "content": f"[背景信息]\n{_clean(req.context)}"})
        messages.append({"role": "assistant", "content": "好的，我已了解背景信息。"})
    messages.append({"role": "user", "content": _clean(req.message)})
```

**为什么需要**：ChromaDB 或 Redis 中可能存储了含有 Unicode 代理字符（`\uD800`-`\uDFFF`）的数据，这些字符在 Python 字符串中合法但无法编码为 UTF-8。`encode("utf-8", errors="ignore")` 会静默丢弃这些字符，防止 API 调用时崩溃。

## 八、环境自适应：第三方 API 兼容

**文件**：`core/intent_recognizer.py` 第 92-106 行

```python
def __init__(self, api_key, base_url=None, model="claude-3-5-sonnet-20241022", ...):
    kwargs = {"api_key": api_key}
    if base_url:
        kwargs["base_url"] = base_url
    self.client = AsyncAnthropic(**kwargs)
    self.model  = model

    # 第三方兼容 API（如 DeepSeek）不支持 Voyage Embeddings，禁用 embedding 策略
    self._embedding_enabled = not bool(base_url)
```

**设计思路**：通过一个 `base_url` 参数实现两种模式的自动切换：

| 模式 | 条件 | 行为 |
|------|------|------|
| 官方 Anthropic | `base_url` 为空 | 三路融合（LLM 70% + Embedding 20% + Pattern 10%） |
| 第三方兼容 | `base_url` 有值 | 两路融合（LLM 85% + Pattern 15%），跳过 Embedding |

调用方不需要关心底层差异，只需传入环境变量即可。
